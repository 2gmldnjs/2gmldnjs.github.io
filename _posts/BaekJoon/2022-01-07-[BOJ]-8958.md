```python
import sys
n=int(input())

for i in range(n):
    oc= 0
    oc_tot=0
    a= sys.stdin.readline()
    for j in range(len(a)):
        if a[j]=='O':
            oc+=1
        else:
            oc=0
        oc_tot+=oc
        
    print(oc_tot)
```
