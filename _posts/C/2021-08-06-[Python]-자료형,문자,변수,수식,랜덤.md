# 연산자

```python
#연산자
print(1+1)
print(3-2)
print(5*2)
print(6/3) # 2.0

print(2**3) #2^3
print(5%3) #나머지
print(5//3) #몫

print(3==3) #True
print(4==2) #False
print(3+4 ==7) #True
print(1!= 3) #True
print(not(1 != 3)) #not 붙으면 반대, 따라서 False
print((3>0) and (3<5)) # 양쪽 모두 True 면 True,따라서 true
print((3>0) & (3<5)) # and 와 & 같은 의미
print((3>0) or (3>5)) # 한쪽이라도 True 면 True,따라서 true
print((3>0) | (3>5)) # or 와 | 같은 의미
print(5>4>3) # True
print(5>4>7) # False
print() 
```

# 퀴즈

